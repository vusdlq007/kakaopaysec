create table "T_USER"
(
 USER_ID bigint generated by default as identity,
 "NAME" varchar(255),
 "AGE" int,
 "CREATED_AT" DATETIME,
 primary key (USER_ID)
);


create table "T_ACCOUNT"
(
 "USER_ID" bigint,
 "ACCOUNT_NUMBER" varchar(255),
 "CREATED_AT" DATETIME,
 foreign key (USER_ID)
 references T_USER(USER_ID) on update CASCADE,
 primary key (USER_ID),
 unique (ACCOUNT_NUMBER)
);

create table "T_ACCOUNT_DETAIL"
(
 "ACCOUNT_NUMBER" varchar(255),
 "IS_DEPOSIT" varchar(1),
 "DEPOSIT_AMOUNT" bigint,
 "CREATED_AT" DATETIME,
 foreign key (ACCOUNT_NUMBER)
 references T_ACCOUNT(ACCOUNT_NUMBER) on update CASCADE,
 primary key (ACCOUNT_NUMBER),
 unique (ACCOUNT_NUMBER)
);
